1. Перейдиет в папку html и выведите список всех папок(без файлов)

cd/var/www/html/ # переходим в нужный каталог
find . -maxdepth 1 -type d  # отбражаем только директории (d) в текущем (maxdepth 1) каталоге (в том числе скрытые)

2. Удалите пустые строки из файла error.txt

cd/var/www/html/logs/ # переходим в нужный каталог
sed '/^$/d' error.txt # командой sed, необходимым регулярным выражением удаляем пустые строки из указаного файла

3. Выведите список файлов с расширением .jpg в папке images

cd/var/www/html/images/ # переходим в нужный каталог
ls | grep '\.jpg$' # командами ls и grep, а также указанием нужного расширения получаем нужный список

4. Напишите bash скрипт который переименует все файлы в папке logs с .txt на .log

cd/var/www/html/logs/ # переходим в нужный каталог

for f in *.txt; do
mv -- "$f" "${f%.txt}.log"
done

# В основе скрипта команда mv,которая переименует файлы, через цикл for укажем паттерн по которому искать файлы с .txt

5. Модифицируйте права доступа и владельца папки images следующим образом: Пользователь www-data и группа www-data имеют чтение/запись на всю дерикторию и на файлы внутри дериктории остальные не имеют доступа к папке и файлам.

sudo chown www-data:www-data cd/var/www/html/images/ # меняем владельца и задаем группу
cd/var/www/html/images/ # переходим в нужный каталог
chmod 660 # 6 - чтение(4) и запись (2) для владельца и группы, 0 - нет доступа для всех остальных

6. Удалите все содержимое в файле access.txt не удаляя сам файл.

echo -n > /var/www/html/logs/access.tx # очистка содержимого файла с помощью команды echo и указанием пути к нему

7. Выведите список файлов в папке images размер которых более 2 мегабайт

find /var/www/html/images -size +2048k # команда find, указываем нужный каталог и нужный размер, например, в килобайтах
